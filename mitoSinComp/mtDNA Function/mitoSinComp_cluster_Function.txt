#Defining quality check
def quality_check(gray, coordinates, db, percent = 0.9):
    # for the items in unique list, find the coordinates
    unique_coords = []

    # find the labels which only occur once
    unique_list = np.unique(db.labels_)
    only_once_list = [i for i in unique_list if list(db.labels_).count(i) == 1]

    for i in only_once_list:
        unique_coords.append(coordinates[db.labels_ == i])

    # using the coordinate as the center, draw a 3x3 square around the array and find the mean pixel intensity of that square
    # if the mean pixel intensity of that square is below 90% of the pixel intensity of the center, remove the coordinate

    remove_list = []
    remove_list_label = []

    for j, i in enumerate(unique_coords):
        x = i[0][0]
        y = i[0][1]
        mean_int = (gray[x-1:x+2, y-1:y+2].mean()) 
        if mean_int < percent * gray[x, y]: 
            # 1 == 1
            remove_list.append(i)
            remove_list_label.append(only_once_list[j])

    if -1 in db.labels_:
        remove_list_label.append(-1)

    return remove_list_label

def plot_foci(labels, rll, gray, binary, coordinates, target_label=None):
    pixel_count_list = []
    clustered = np.zeros_like(binary)

    fig, axs = plt.subplots(1, 3, figsize=(15, 5))
    axs[0].imshow(clustered, cmap='tab20')
    axs[1].imshow(gray, cmap='gray')
    axs[2].imshow(binary, cmap='gray')

    for label in np.unique(labels):
        if label == -1 or label in rll:
            continue
        if target_label is not None and label != target_label:
            continue

        coords = coordinates[labels == label]
        if len(coords) < MIN_SAMPLES:
            continue

        clustered[coords[:, 0], coords[:, 1]] = 100
        pixel_count_list.append([label, len(coords)])

        y_mean, x_mean = coords[:, 0].mean(), coords[:, 1].mean()

        axs[0].text(x_mean, y_mean, str(label), color='white', fontsize=6)
        axs[1].text(x_mean, y_mean, str(label), color='white', fontsize=6)
        axs[1].scatter(coords[:, 1], coords[:, 0], marker='x', color=cm.tab20(label), alpha=ALPHA, s=1)
        axs[2].scatter(coords[:, 1], coords[:, 0], marker='x', color=cm.tab20(label), alpha=ALPHA, s=1)

    plt.tight_layout()
    return pixel_count_list

from scipy.spatial import distance

# Input CSV
foci_csv = "*/filtered_coords_thresh_2_min_sample_6_cluster_size.csv" #Replace * with file location
structure_csv = "*/AnalysisFiles_cell/full_Table_cell.csv" #Replace * with file location
#output to be saved as..
output_plot_csv = "*/plot.csv" #Replace * with file location
output_noplot_csv = "*/noplot.csv" #Replace * with file location

# Scale factors (µm per pixel)
XY_SCALE = 0.099  # µm per pixel
Z_SCALE = 0.5     # µm per slice
MIN_DISTANCE = 4 # distance threshold (in pixel units)
CLUSTER_COL = "cluster_num"

# LOAD DATA

foci_df = pd.read_csv(foci_csv)
structure_df = pd.read_csv(structure_csv)

# Ensure numeric
foci_df[['x_pixel', 'y_pixel', 'z_pixel']] = foci_df[['x_pixel', 'y_pixel', 'z_pixel']].apply(pd.to_numeric)

# CONVERT STRUCTURE COORDINATES TO PIXEL SPACE

# Divide µm values by µm-per-pixel scale to get pixel-equivalent positions
structure_df['x_pixel'] = (structure_df['x_x'] / XY_SCALE).round(0)
structure_df['y_pixel'] = (structure_df['y_x'] / XY_SCALE).round(0)
structure_df['z_pixel'] = (structure_df['z_x'] / Z_SCALE).round(0)

# Save structure_df with new pixel-based columns for reference
structure_df.to_csv("*/AnalysisFiles_cell/structure_converted_pixels.csv", index=False) #Replace * with file location
print("Saved structure_df with converted pixel coordinates.")
structure_df= structure_df.rename(columns={'Unnamed: 0': 'Index'})  # Ensure there is an Index column

# COMPUTE PAIRWISE DISTANCES

coords_foci = foci_df[['x_pixel', 'y_pixel', 'z_pixel']].to_numpy()
coords_struct = structure_df[['x_pixel', 'y_pixel', 'z_pixel']].to_numpy()    

# Calculate pairwise Euclidean distances
distances = distance.cdist(coords_foci, coords_struct)

# Find minimum distance per foci pixel
foci_df['min_distance'] = distances.min(axis=1)
foci_df['Index'] = distances.argmin(axis=1)

# CLUSTER-LEVEL MAPPING

cluster_min = foci_df.groupby(CLUSTER_COL)['min_distance'].min()
mapped_clusters = cluster_min[cluster_min <= MIN_DISTANCE].index
unmapped_clusters = cluster_min[cluster_min > MIN_DISTANCE].index

foci_df['plot'] = np.where(foci_df[CLUSTER_COL].isin(mapped_clusters), 'PLOT', 'NO_PLOT')

# Split datasets
foci_plot = foci_df[foci_df['plot'] == 'PLOT']
foci_noplot = foci_df[foci_df['plot'] == 'NO_PLOT']
